import { createStore, combineReducers, applyMiddleware } from 'redux'; import thunk from 'redux-thunk'; import * as reducers from './reducers/index';

const logger = store => next => action => { if (typeof action !== "function") { console.log('dispatching', action); } return next(action); };

export default function store(viewer) { const middleware = [thunk.withExtraArgument(viewer)]; return createStore(combineReducers(reducers), applyMiddleware(...middleware)); }
