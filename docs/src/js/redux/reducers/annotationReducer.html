Annotation Reducer

This is where we record the state of the annotations. This will include adding the annotations initially and also recording the state when the annotations are being viewed.

import { ACTIVE_ANNOTATION } from '../actions/annotationActions'; const default_state = { selected: {}, list: [ { name: 'first', x: 0.10970329757163497, y: 0.3352769748048794, zoom: 2.6274194716671744, rotation: 333 }, { name: 'second', x: 0.25, y: 0.34, zoom: 4.5, rotation: 315 }, { name: 'third', x: 0.8969657248691317, y: 0.7033570144622934, zoom: 2.724108508224526, rotation: 298 } ] }; export default function annotationReducer(state = default_state, action) { switch (action.type) { case ACTIVE_ANNOTATION: return Object.assign({}, state, { selected: action.payload }); default: return state; } }
